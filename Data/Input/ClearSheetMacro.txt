Sub ClearSheet(xlSheets As Variant, keepFirstRow As Boolean)
	Dim offset As Integer
	
	If keepFirstRow = True Then
		offset = 1
	Else
		offset = 0
	End If
	
	For Each xlSheet in xlSheets
		Sheets(xlSheet).UsedRange.Offset(offset).ClearContents
	Next
	
	'Dim xlSheet As Worksheet
	'Set xlSheet = Workbooks(workbook).Sheets(sheetName)
	'xlSheet.Range("A2:H" & .Cells(.Rows.Count, "A").End(xlUp).Row).ClearContents
End Sub
